<script>
(function () {
	var nav = document.getElementById("via-mobile-menu");
	var toggle = document.querySelector(".via-mobile-menu-toggle");
	var closeButton = document.querySelector(".via-mobile-menu-close");
	if (!nav || !toggle || !closeButton) {
		return;
	}
	function setOpenState(open) {
		if (open) {
			nav.classList.add("via-mobile-open");
			document.body.classList.add("via-mobile-menu-open");
			toggle.setAttribute("aria-expanded", "true");
		} else {
			nav.classList.remove("via-mobile-open");
			document.body.classList.remove("via-mobile-menu-open");
			toggle.setAttribute("aria-expanded", "false");
		}
	}
	toggle.addEventListener("click", function () {
		setOpenState(!nav.classList.contains("via-mobile-open"));
	});
	closeButton.addEventListener("click", function (event) {
		event.preventDefault();
		setOpenState(false);
	});
	Array.prototype.forEach.call(nav.querySelectorAll("a"), function (link) {
		link.addEventListener("click", function () {
			setOpenState(false);
		});
	});
	document.addEventListener("keydown", function (event) {
		if (event.key === "Escape") {
			setOpenState(false);
		}
	});
})();
</script>
